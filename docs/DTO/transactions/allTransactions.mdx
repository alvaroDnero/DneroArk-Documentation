# Get All The Transactions

| Name                  | Type       |   Description                                                                                  |
|-----------------------|------------|-----------------------------------------------------------------------------------------------|
| `transactionId`       | `long`     | Unique identifier of the transaction notification.                                           |
| `interactionType`     | `int`      | Possible values: `0` for sent or `1` received.                                                        |
| `amount`              | `string`   | The amount of money involved in the transaction.                                             |
| `coinStatus`          | `int`      | Coin status must not be zero. For detailed information, see the [Status Codes page.](/docs/coinsStatuses)|
| `expirationDate`      | `DateTime` | Expiration date of the coin. |
| `capturedDate`        | `DateTime` | Date when the coin was captured. |
| `createDate`          | `DateTime` | Date when the coin was created. |
| `user`                | `object`   | If `interactionType` is `0` it’s the recipient's information, ; if `1`, it’s the sender's information. See [ User Object Details.](/docs/Objects/userObject) |

```c#
public class TransactionDto
{
    /// <summary>
    /// Unique identifier of the transaction notification.
    /// </summary>
    public long TransactionId { get; set; }

    /// <summary>
    /// Possible values: 0 for sent or 1 for received.
    /// </summary>
    public int InteractionType { get; set; }

    /// <summary>
    /// The amount of money involved in the transaction.
    /// </summary>
    public string Amount { get; set; }

    /// <summary>
    /// Coin status must not be zero.
    /// </summary>
    public int CoinStatus { get; set; }

    /// <summary>
    /// Expiration date of the coin.
    /// </summary>
    public DateTime? ExpirationDate { get; set; }

    /// <summary>
    /// Date when the coin was captured.
    /// </summary>
    public DateTime? CapturedDate { get; set; }

    /// <summary>
    /// Date when the coin was created.
    /// </summary>
    public DateTime CreateDate { get; set; }

    /// <summary>
    /// User details based on interaction type.
    /// </summary>
    public UserDto User { get; set; }
}


public class PaginatedTransactionResponseDto
{
    /// <summary>
    /// The current page number being returned (starting from 1).
    /// </summary>
    public int CurrentPage { get; set; }

    /// <summary>
    /// The total number of pages available based on the page size and total records.
    /// </summary>
    public int TotalPages { get; set; }

    /// <summary>
    /// The number of items per page in the paginated results.
    /// </summary>
    public int PageSize { get; set; }

    /// <summary>
    /// The total number of elements available.
    /// </summary>
    public int TotalElements { get; set; }

    /// <summary>
    /// A list containing all the transactions for the requested page.
    /// </summary>
    public List<TransactionDto> Transactions { get; set; }
}
```
