import HttpMethodTag from '@site/src/components/HttpMethodTag/HttpMethodTag';
export const apiPath = ' /DneroArk/login/logout';
export const method = 'POST';





# Title 


<div className="highlight-box">
  <HttpMethodTag method={method} /> {apiPath}
</div>





Delete the tokens and log the logout activity on the device.

### Header

| Key      | Value     |
|-------------|----------|
| `Authorization` | Bearer `{{accessToken}}`|






---
### JSON Response: Success (200)
```JSON
{
  "code": "LOGOUT_SUCCESS",
  "message": "The session was successfully terminated."
}
```
---
### JSON Response: Error

- 400 Bad Request
```JSON
{
  "code": "INVALID_REQUEST",
  "message": "The authorization token is required."
}
```

- 401 Unauthorized

```JSON
{
  "code": "TOKEN_INVALID",
  "message": "The authorization token is invalid or has expired."
}
```


- 403 Forbidden

```JSON
{
  "code": "TOKEN_FORBIDDEN",
  "message": "You do not have permission to log out from this session."
}

```


- 500 Internal Server Error

```JSON
{
  "code": "INTERNAL_SERVER_ERROR",
  "message": "An unexpected error occurred while validating the OTP. Please try again later."
}

```
